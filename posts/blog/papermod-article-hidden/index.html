PaperMod Article Hidden

Hugo Theme PaperMod Add Hidden Article Function

1. Edit Article YAML#

Add hidden: true in the article yaml section

---
title: "Article Title"
date: 
lastmod: 
author: 
hidden: true
---

2. Hide Article IN Article List Page#

File Path : \layouts_default\list.html

Insert {{ if not .Params.hidden }}{{- end }} to the specified position


{{ if not .Params.hidden }}<!-- hidden-md-edit begin -->

<article class="{{ $class }}">
  <div class="post-info">
    <header class="entry-header">
      <h2>
        {{- .Title }}
        {{- if .Draft }}<sup><span class="entry-isdraft">&nbsp;&nbsp;[draft]</span></sup>{{- end }}
      </h2>
    </header>
    {{- if (ne (.Param "hideSummary") true) }}
    <section class="entry-content">
      <p>
        {{- if .Description }}
        {{ .Description }}
        {{- end }}
        <!-- {{ .Summary | plainify | htmlUnescape }} -->
        {{ if .Truncated }}...{{ end }}
      </p>
    </section>
    {{- end }}
    {{- if not (.Param "hideMeta") }}
    <footer class="entry-footer">
      {{- partial "post_meta.html" . -}}
    </footer>
    {{- end }}
  </div>
  {{- $isHidden := (.Site.Params.cover.hidden | default .Site.Params.cover.hiddenInList) }}
  {{- partial "cover.html" (dict "cxt" . "IsHome" true "isHidden" $isHidden) }}
  <a class="entry-link" aria-label="post link to {{ .Title | plainify }}" href="{{ .Permalink }}"></a>
</article>
{{- end }}

{{- end }}<!-- hidden-md-edit end -->

3. Archives#

3.1 Hidden In Archives#

File Path : \layouts\_default\archives.html

<div class="archive-posts">
    {{- range .Pages }}
    {{- if not .Params.hidden }} <!-- Hidden-In-Archive Begin -->
    {{- if eq .Kind "page" }}
    <div class="archive-entry">
    <h3 class="archive-entry-title">
        {{- .Title | markdownify }}
        {{- if .Draft }}<sup><span class="entry-isdraft">&nbsp;&nbsp;[draft]</span></sup>{{- end }}
    </h3>
    <div class="archive-meta">
        {{- partial "post_meta.html" . -}}
    </div>
    <a class="entry-link" aria-label="post link to {{ .Title | plainify }}" href="{{ .Permalink }}"></a>
    </div>
    {{- end }}
    {{- end }}
    {{- end }}<!-- Hidden-In-Archive End -->
</div>
